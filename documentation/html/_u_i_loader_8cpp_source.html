86  static_cast<int8_t>(std::atoi(loader->getValueByKey("ROLLS", mv::constants::loader::CONFIG_MODE::TECHNICALITIES).c_str())),

89

91 
92  float delta_x = windowSize.x / (crystalAmmount.x + 1);
93  float delta_y = (windowSize.y*resolutionFactor) / (crystalAmmount.y + 1);

110 void UILoader::checkDimensions(std::shared_ptr<mv::Entity> exampleSymbol, const Vector2<float>& delta, const std::shared_ptr<Loader>& loader)
111 {
112  auto tolerance = std::atof(loader->getValueByKey("TOLERANCE", mv::constants::loader::CONFIG_MODE::TECHNICALITIES).c_str());
113  std::vector<std::string> errors;
114  bool flag_error = false;
115  if ( exampleSymbol->getComponent<ProperBody>()->getRect().w*(1 + tolerance) > delta.x )
116  {
117  errors.emplace_back(mv::constants::error::UI::TOO_CLOSE_HORIZONTALLY);
118  flag_error = true;
119  }
120 
121  if ( exampleSymbol->getComponent<ProperBody>()->getRect().h*(1 + tolerance) > delta.y )
122  {
123  errors.emplace_back(mv::constants::error::UI::TOO_CLOSE_VERTICALLY);
124  flag_error = true;
125  }
126 
127  if ( flag_error )
128  {
129  for(auto&var:errors )
130  {
131  var.append(" ");
132  var.append(mv::constants::error::UI::SOLUTION);

134  }
135  }
